<script>
async function loadAdopters() {
    const res = await fetch("/adopters");
    const data = await res.json();
    const container = document.getElementById("list");
    if (!data.length) {
        container.innerHTML = "<p>No adopters yet.</p>";
        return;
    }
    container.innerHTML = data.map(a => `
        <div class="pet-card">
            <h3>${a.adopter_name}</h3>
            <p><strong>Phone:</strong> ${a.phone}</p>
            <p><strong>Address:</strong> ${a.address}</p>
            <p><strong>Pet:</strong> ${a.pet_name}</p>
        </div>
    `).join("");
}
loadAdopters();
</script>
