<script>
async function loadPets() {
    const res = await fetch("/pets");
    const data = await res.json();
    const container = document.getElementById("pets");
    if (!data.length) {
        container.innerHTML = "<p>No pets yet.</p>";
        return;
    }
    container.innerHTML = data.map(p => `
        <div class="pet-card">
            <img src="${p.image}" alt="${p.name}">
            <h3>${p.name}</h3>
            <p>Type: ${p.type}</p>
            <p>Age: ${p.age}</p>
            <p>${p.adopted ? "<strong>Adopted</strong>" : ""}</p>
        </div>
    `).join("");
}
loadPets();
</script>
